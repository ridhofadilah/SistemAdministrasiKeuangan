12     public class ControllerLogin implements ActionListener {
13         private Aplikasi model;
14         private Login view;
15         private Database data = new Database();
16     
17         ControllerLogin(Aplikasi model) {
18                 this.model=model;
19                 data.connect();
20                 view = new Login();
21                 view.setVisible(true);
22                 view.addListener(this);
23         }
24     
25         @Override
26         public void actionPerformed(ActionEvent ae) {
27             Object source = ae.getSource();
28             if (source == view.getBtnLogin()){
29                 String uname = view.getTfUsername();
30                 String pw = view.getTfPassword();
31                 if (uname.equals("")|| pw.equals("")){
32                     JOptionPane.showMessageDialog(null,"Fill the username/password");
33                 } else if (uname.equals("admin") && pw.equals("admin")){
34                     new ControllerMenuAdmin(model);
35                     view.dispose();
36                 } else if (uname.equals("wakilrektor") && pw.equals("wakilrektor")){
37                     new ControllerMenuWakilRektor(model);
38                     view.dispose();
39                 } else if (model.findMahasiswa(uname,pw) != null){
40                     Mahasiswa m = model.findMahasiswa(uname, pw);
41                     new ControllerMenuMahasiswa(model, m);
42                     view.dispose();
43                 } else if (model.findFakultas(uname,pw) != null){
44                     Fakultas f = model.findFakultas(uname, pw);
45                     new ControllerMenuFakultas(model, f);
46                 } else {
47                     JOptionPane.showMessageDialog(null, "Sorry user is not in database!");
48                 }
49             }
50         }
