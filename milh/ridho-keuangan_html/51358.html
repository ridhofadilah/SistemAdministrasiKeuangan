14     import java.awt.event.ActionListener;
15     import javax.swing.JOptionPane;
16     
17     /**
18      *
19      * @author User
20      */
21     public class ControllerLogin implements ActionListener{
22         private Aplikasi model;
23         private Login view;
24         private Database data = new Database();
25     
26         public ControllerLogin(Aplikasi model) {
27                 this.model=model;
28                 data.connect();
29                 view = new Login();
30                 view.setVisible(true);
31                 view.addListener(this);
32         }
33     
34         @Override
35         public void actionPerformed(ActionEvent ae) {
36             Object source = ae.getSource();
37             if (source == view.getBtnLogin()){
38                 String uname = view.getTfUsername();
39                 String pw = view.getTfPassword();
40                 if ("".equals(uname)|| "".equals(pw)){
41                     JOptionPane.showMessageDialog(null,"Fill the username/password");
42                 } else if ("admin".equals(uname) && "admin".equals(pw)){
43                     new ControllerMenuAdmin(model);
44                     view.dispose();
45                 } else if ("wakilrektor".equals(uname) && "wakilrektor".equals(pw)){
46                     new ControllerMenuWakilRektor(model);
47                     view.dispose();
48                 } else if (model.findMahasiswa(uname,pw) != null){
49                     Mahasiswa m = model.findMahasiswa(uname, pw);
50                     new ControllerMenuMahasiswa(model, m);
51                     view.dispose();
52                 } else if (model.findFakultas(uname,pw) != null){
53                     Fakultas f = model.findFakultas(uname, pw);
54                     new ControllerMenuFakultas(model, f);
55                 } else {
56                     JOptionPane.showMessageDialog(null, "Sorry user is not in database!");
57                 }
58             }
59         }
60     }
